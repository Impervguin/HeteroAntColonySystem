\begin{itemize}
	\item $N_g$ -- количество поколений;
	\item $\mu_r$ -- средняя длина полученного маршрута по $R$ запускам;
	\item $\sigma_r$ -- среднеквадратичное отклонение длины полученного маршрута по $R$ запускам;
	\item $ITB_{1/2}$ -- медиана количества итераций до получения лучшего решения;
	\item $\mu_{t}$ -- среднее время работы алгоритма в миллисекундах.
\end{itemize}
