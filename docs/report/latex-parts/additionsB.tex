\ssr{ПРИЛОЖЕНИЕ Б}

Диск с рабочими материалами со следующей структурой папок:
\begin{verbatim}
	api -- каталог для HTTP-API
	cmd -- каталог с точками входа (main-пакеты)
	docs -- каталог с РПЗ ВКР
	internal/ -- каталог с кодом метода
	  core -- каталог с ядром метода
	  observers -- католог с наблюдателями
	  strategies -- каталог с реализациями стратегий
	pkg -- публичные пакеты для переиспользования
	scripts -- вспомогательные скрипты (сбор и анализ)
	tsp -- каталог с файлами TSPLIB
	web -- веб-интерфейс/фронтенд (HTML/CSS/JS/шаблоны)
	go.mod -- модульный файл Go (зависимости)
	go.sum -- контрольные суммы зависимостей Go
\end{verbatim}
